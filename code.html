                        // Logic using If Statments 
                        function fuzzybuzz(fizzyValue, buzzValue) {

                            // Arry of numbers 
                            let returnArray = [];

                            // Looping through the numbers 
                            for (let i = 1; i <= 100; i++) {

                                if (i % fizzyValue == 0 && i % buzzValue == 0) {
                                    returnArray.push('FizzyBuzz');
                                } else if (i % fizzyValue == 0) {
                                    returnArray.push("Fizzy");
                                } else if (i % buzzValue == 0) {
                                    returnArray.push("Buzz");
                                }
                                // Pushing to the numbers to numberValues
                                returnArray.push(i);
                            }
                            return returnArray;
                        }

                        function fuzzybuzzB(fizzyValue, buzzValue) {

                            // Arry of numbers 
                            let returnArray = [];
                            let fizzy = false;
                            let buzz = false;

                            // Looping through the numbers 
                            for (let i = 1; i <= 100; i++) {

                                // New Code Using Switch statments
                                fizzy = i % fizzyValue == 0;
                                buzz = i % buzzValue == 0;

                                switch (true) {
                                    case fizzy && buzz: {
                                        returnArray.push("FizzyBuzz");
                                        break;
                                    }
                                    case fizzy: {
                                        returnArray.push("Fizzy");
                                        break;
                                    }
                                    case buzz: {
                                        returnArray.push("Buzz");
                                        break;
                                    }
                                    default: {
                                        // Pushing to the numbers to numberValues
                                        returnArray.push(i);
                                        break;
                                    }
                                }
                            }
                            return returnArray;
                        }

Fizzy Buzzy is Presented using Prism.js

The FizzBuzz algorithm is a classic coding test given in coding interviews. Pseudo Code:

1. Print integers 1 to N.
2. Print “Fizzy” if an integer is divisible by the fizzy Value.
3. Print “Buzz” if an integer is divisible by the buzz Value.
4. Print “FizzBuzz” if an integer is divisible by both fizzy Value and buzz Value.
